import { Hooks, Center, Viewport } from "tchin-react-components";

const { useBackground } = Hooks;

export default () => {
    const BGViewport = useBackground(Viewport, "path/to/image.jpg", {
        parralax: false,
        blurAmount: 5,
    });
    return (
        <BGViewport>
            <Center>Viewport container with background</Center>
        </BGViewport>
    );
};
